About JoyVet CDS

JoyVet Clinical Diagnosis Support v1.0

A fully offline-capable, browser-based veterinary clinical decision support tool. Built for use in busy clinical environments. Requires no internet connection — simply open index.html to start.

Knowledge Base

Covers 80+ conditions across Dermatology, Internal Medicine, Oncology, OB/GYN, Neurology, Emergency, Musculoskeletal, and Exotic Animal medicine. All clinical data is based on evidence-based veterinary standards.

Disclaimer

This tool is a clinical decision support aid only. All diagnoses, treatments, and clinical decisions must be made by a licensed veterinarian using their clinical judgment. Drug dosages are guidelines — always verify with current formularies. Not a substitute for professional veterinary assessment.

Species Supported

Dog · Cat · Rabbit · Bird (Avian) · Exotic/Small Mammal

How to Use (Offline)

1. Copy the JoyVet-CDS folder to your device (USB, local drive, LAN share).
2. Double-click index.html to open in any modern browser (Chrome, Firefox, Edge, Safari).
3. Case data saved locally. Use Cloud Sync in Case History to save online.

📁 Google Drive Backup (One-Time Setup)

Automatically saves every case to your personal Google Drive, organised by Species → Client → Patient.

  1. Go to script.google.com → click New project
  2. Delete any existing code, paste the script below, click 💾 Save
  3. Click Deploy → New deployment → Web app
  4. Set Execute as: Me and Who has access: Anyone → Deploy
  5. Copy the Web app URL and paste it in the box below → Save
function doPost(e) {
  try {
    var payload = JSON.parse(e.postData.contents);
    var data = payload.data || payload;
    var species = cap(data.species  || 'Unknown');
    var client  = clean(data.ownerName   || 'Unknown Client');
    var patient = clean(data.patientName || 'Unknown Patient');
    var date    = data.visitDate || fmtDate();
    var root    = getFolder('JoyVet CDS Records', DriveApp.getRootFolder());
    var spF     = getFolder(species,  root);
    var clF     = getFolder(client,   spF);
    var ptF     = getFolder(patient,  clF);
    if (payload.type === 'html' && payload.html) {
      var fname = date + '_' + patient + '_record.html';
      var ex = ptF.getFilesByName(fname);
      if (ex.hasNext()) { ex.next().setContent(payload.html); }
      else { ptF.createFile(fname, payload.html, MimeType.HTML); }
    } else {
      var fname2 = date + '_' + patient + '_record.json';
      var ex2 = ptF.getFilesByName(fname2);
      var content = JSON.stringify(data, null, 2);
      if (ex2.hasNext()) { ex2.next().setContent(content); }
      else { ptF.createFile(fname2, content, MimeType.PLAIN_TEXT); }
    }
    return ContentService.createTextOutput('{"status":"ok"}')
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput('{"status":"error","msg":"'+err.message+'"}')
      .setMimeType(ContentService.MimeType.JSON);
  }
}
function getFolder(name, parent) {
  var f = parent.getFoldersByName(name);
  return f.hasNext() ? f.next() : parent.createFolder(name);
}
function cap(s) { return s.charAt(0).toUpperCase()+s.slice(1).toLowerCase(); }
function clean(s) { return s.replace(/[\\\/\:\*\?\"\<\>\|]/g,'_').trim().substring(0,50)||'Unknown'; }
function fmtDate() { return Utilities.formatDate(new Date(),'GMT+7','yyyy-MM-dd'); }
